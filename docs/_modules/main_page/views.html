    template_name = 'main_page/index.html'

        context['items'] = Development.objects.all().order_by('-development_date')

def feedback_update(request, item_id):
    '''
    saves the new feedback data to the database

    :type request: object
    :param request: the request sent from the browser

    :type item_id: integer
    :param item_id: the item selected by the user which returns its pk to be search in the db
    '''
    if(request.method == 'POST'):
        feedback_data = request.POST.dict()
        # needs valid check
        username = feedback_data.get("feedback_user_name")
        email = feedback_data.get("feedback_user_email")
        reason = feedback_data.get("feedback_reason")
        content = feedback_data.get("feedback_content")
        resovled = feedback_data.get("feedback_resovled")

        date = Feedback.objects.get(pk=item_id)
        db_feedback = Feedback(   pk=item_id, 
                                        feedback_user_name=username,
                                        feedback_user_email=email,
                                        feedback_reason=reason,
                                        feedback_content=content,
                                        feedback_resovled=resovled,
                                        feedback_date=date.feedback_date)
        db_feedback.save()

        template_name = loader.get_template('main_page/success.html')
        context_object_name = {}

        return HttpResponse(template_name.render(context_object_name, request))

[docs]def feedback_new(request):
    '''
    updates development data to the database

    :type request: object
    :param request: the request sent from the browser

    '''
    if (request.method == 'POST'):

        form = FeedbackForm(request.POST)

        if form.is_valid():
            feedback_user_name = form.cleaned_data['feedback_user_name']
            feedback_user_email = form.cleaned_data['feedback_user_email']
            feedback_reason = form.cleaned_data['feedback_reason']
            feedback_content = form.cleaned_data['feedback_content']

            Feedback.objects.create( feedback_user_name=feedback_user_name,
                                        feedback_user_email=feedback_user_email,
                                        feedback_reason=feedback_reason,
                                        feedback_content=feedback_content,
                                        ).save()

            return HttpResponseRedirect('/')

    else:
        form = FeedbackForm()

    return render(request, 'main_page/feedback_form.html', {'form': form})

def development_update(request, item_id):
    '''
    updates development data to the database

    :type request: object
    :param request: the request sent from the browser

    :type item_id: integer
    :param item_id: the item selected by the user which returns its pk to be search in the db
    '''
    if (request.method == 'POST'):

        form = DevelopmentForm(request.POST)

        if form.is_valid():
            development_username = form.cleaned_data['development_username']
            development_project = form.cleaned_data['development_project']
            development_title = form.cleaned_data['development_title']
            development_content = form.cleaned_data['development_content']

            # if it has no id then create a new entry into the db
            if item_id == '':
                Development.objects.create( development_username=development_username,
                                            development_project=development_project,
                                            development_title=development_title,
                                            development_content=development_content
                                            ).save()

                return HttpResponseRedirect('/')
            # if it has an id then update the previous entry
            else:
                date = Development.objects.get(pk=item_id)

                db_development = Development(   pk=item_id, 
                                                development_username=development_username,
                                                development_project=development_project,
                                                development_title=development_title,
                                                development_content=development_content,
                                                development_date=date.development_date)
                db_development.save()

                return HttpResponseRedirect('/')
    else:
        form = DevelopmentForm()

    return render(request, 'main_page/development_form.html', {'form': form})
